1""" 

2Domain 6: Courtroom-Grade Visualization Specifications 

3======================================================== 

4 

5Specifications for charts that survive courtroom scrutiny, including: 

6 - Okabe-Ito colorblind-safe palette (WCAG AA compliant) 

7 - ReportLab Platypus layout architecture (no text/image overlap) 

8 - Standard sizing for letter paper with 1-inch margins 

9 - FRE 902(13)/(14) compliant evidence labeling 

10 - Data source attribution on every chart 

11 

12Chart container: single-cell ReportLab Table with full border control 

13 - 468pt wide x 288pt tall (6.5" x 4") for full-width charts 

14 - 0.75pt borders in dark gray (#333333) 

15 - 12pt internal padding, 18pt external spacing 

16 

17Key anti-overlap flowables: 

18 - KeepTogether: chart title + chart drawing + source note on same page 

19 - CondPageBreak: trigger page break if insufficient vertical space 

20 - Spacer: explicit vertical gaps between chart containers 

21 

22References: 

23 - Okabe & Ito (2008) Color Universal Design palette 

24 - WCAG 2.1 AA: 4.5:1 contrast minimum for text, 3:1 for graphics 

25 - ISO 19005 (PDF/A archival compliance) 

26 - FRE 902(13), 902(14) (self-authentication for electronic evidence) 

27"""

61@dataclass 

62class ChartSpec: 

63 """Specification for a single courtroom-grade chart.""" 

64 chart_id: str 

65 title: str # Format: "Figure [N]: [Descriptive Title]" 

66 chart_type: str # bar, line, scatter, network, treemap, timeline 

67 width_pt: float = 468 # 6.5 inches at 72 dpi 

68 height_pt: float = 288 # 4 inches at 72 dpi 

69 border_width_pt: float = 0.75 

70 border_color: str = '#333333' 

71 internal_padding_pt: float = 12 

72 external_spacing_pt: float = 18 

73 

74 # Typography 

75 title_font: str = 'Times-Bold' 

76 title_size_pt: float = 12 

77 axis_label_font: str = 'Helvetica' 

78 axis_label_size_pt: float = 9 

79 data_label_font: str = 'Helvetica' 

80 data_label_size_pt: float = 8 

81 

82 # Data source attribution (required on every chart) 

83 source_text: str = '' # e.g., "Source: NIKE Form 4 filings, FY2019" 

84 source_font: str = 'Helvetica-Oblique' 

85 source_size_pt: float = 8 

86 

87 # Color palette 

88 colors: Dict[str, str] = field(default_factory=lambda: dict(OKABE_ITO_PALETTE)) 

89 

90 # Y-axis configuration 

91 y_axis_label: str = '' 

92 y_axis_format: str = '' # e.g., '$XXM', '$XXK', '%' 

93 

94 # Render settings 

95 dpi: int = 300 

96 font_type: int = 42 # TrueType for editability 

97 transparent_bg: bool = False 

98 remove_top_spine: bool = True 

99 remove_right_spine: bool = True 

100 

101 def to_dict(self) -> dict: 

102 return { 

103 'chart_id': self.chart_id, 

104 'title': self.title, 

105 'chart_type': self.chart_type, 

106 'dimensions': f'{self.width_pt}pt x {self.height_pt}pt', 

107 'source_text': self.source_text, 

108 'y_axis_label': self.y_axis_label, 

109 'y_axis_format': self.y_axis_format, 

110 'dpi': self.dpi, 

111 }
