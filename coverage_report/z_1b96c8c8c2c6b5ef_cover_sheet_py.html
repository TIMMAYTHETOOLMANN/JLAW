1""" 

2Cover Sheet Formatter - Phase 4 Enhanced 

3========================================= 

4 

5Generates DOJ-grade confidential cover sheet with case metadata using Unicode box drawing. 

6 

7Format matches exact specification with proper case identifier, analysis period, 

8and classification fields. 

9""" 

10 

11from datetime import datetime 

12from typing import Dict, Any 

13from .format_constants import ( 

14 BOX_DOUBLE_HORIZONTAL, 

15 BOX_DOUBLE_VERTICAL, 

16 BOX_DOUBLE_TOP_LEFT, 

17 BOX_DOUBLE_TOP_RIGHT, 

18 BOX_DOUBLE_BOTTOM_LEFT, 

19 BOX_DOUBLE_BOTTOM_RIGHT, 

20 BOX_SINGLE_HORIZONTAL, 

21 BOX_SINGLE_VERTICAL, 

22 BOX_SINGLE_TOP_LEFT, 

23 BOX_SINGLE_TOP_RIGHT, 

24 BOX_SINGLE_BOTTOM_LEFT, 

25 BOX_SINGLE_BOTTOM_RIGHT, 

26 CLASSIFICATION_CONFIDENTIAL, 

27 STANDARD_WIDTH, 

28) 

29 

30 

31class CoverSheetFormatter: 

32 """Formats DOJ-grade cover sheet for prosecutorial dossiers.""" 

33 

34 @staticmethod 

35 def format(case_data: Dict[str, Any]) -> str: 

36 """ 

37 Format cover sheet with case metadata per specification. 

38  

39 Args: 

40 case_data: Dictionary with keys: 

41 - case_id: JLAW case identifier 

42 - company_name: Company under investigation 

43 - cik: SEC CIK number 

44 - generation_date: Report generation timestamp 

45 - dossier_type: Classification (DOJ-GRADE/SEC REFERRAL READY) 

46 - start_date: Analysis period start 

47 - end_date: Analysis period end 

48  

49 Returns: 

50 Formatted cover sheet string with Unicode box drawing 

51 """ 

52 lines = [] 

53 

54 # Double-line outer box header 

55 lines.append(f"{BOX_DOUBLE_TOP_LEFT}{BOX_DOUBLE_HORIZONTAL * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_TOP_RIGHT}") 

56 lines.append(f"{BOX_DOUBLE_VERTICAL}{' ' * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_VERTICAL}") 

57 

58 # Classification header 

59 classification_text = CLASSIFICATION_CONFIDENTIAL 

60 padding = (STANDARD_WIDTH - 2 - len(classification_text)) // 2 

61 lines.append(f"{BOX_DOUBLE_VERTICAL}{' ' * padding}{classification_text}{' ' * (STANDARD_WIDTH - 2 - padding - len(classification_text))}{BOX_DOUBLE_VERTICAL}") 

62 lines.append(f"{BOX_DOUBLE_VERTICAL}{' ' * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_VERTICAL}") 

63 

64 # Inner box for title 

65 inner_box_width = STANDARD_WIDTH - 8 

66 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_TOP_LEFT}{BOX_SINGLE_HORIZONTAL * inner_box_width}{BOX_SINGLE_TOP_RIGHT} {BOX_DOUBLE_VERTICAL}") 

67 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * inner_box_width}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

68 

69 title = "SECURITIES FRAUD FORENSIC DOSSIER" 

70 title_padding = (inner_box_width - len(title)) // 2 

71 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * title_padding}{title}{' ' * (inner_box_width - title_padding - len(title))}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

72 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * inner_box_width}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

73 

74 # Company name and CIK 

75 company_name = case_data.get('company_name', 'N/A') 

76 cik = case_data.get('cik', 'N/A') 

77 company_padding = (inner_box_width - len(company_name)) // 2 

78 cik_text = f"CIK: {cik}" 

79 cik_padding = (inner_box_width - len(cik_text)) // 2 

80 

81 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * company_padding}{company_name}{' ' * (inner_box_width - company_padding - len(company_name))}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

82 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * cik_padding}{cik_text}{' ' * (inner_box_width - cik_padding - len(cik_text))}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

83 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_VERTICAL}{' ' * inner_box_width}{BOX_SINGLE_VERTICAL} {BOX_DOUBLE_VERTICAL}") 

84 lines.append(f"{BOX_DOUBLE_VERTICAL} {BOX_SINGLE_BOTTOM_LEFT}{BOX_SINGLE_HORIZONTAL * inner_box_width}{BOX_SINGLE_BOTTOM_RIGHT} {BOX_DOUBLE_VERTICAL}") 

85 lines.append(f"{BOX_DOUBLE_VERTICAL}{' ' * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_VERTICAL}") 

86 

87 # Metadata table 

88 case_id = case_data.get('case_id', 'N/A') 

89 start_date = case_data.get('start_date', 'N/A') 

90 end_date = case_data.get('end_date', 'N/A') 

91 analysis_period = f"{start_date} — {end_date}" 

92 classification = case_data.get('dossier_type', 'DOJ-GRADE') 

93 generation_date = case_data.get('generation_date', 'N/A') 

94 if isinstance(generation_date, datetime): 

95 generation_date = generation_date.strftime('%Y-%m-%d %H:%M:%S UTC') 

96 

97 lines.append(f"{BOX_DOUBLE_VERTICAL} Analysis Period │ {analysis_period:<52}{BOX_DOUBLE_VERTICAL}") 

98 lines.append(f"{BOX_DOUBLE_VERTICAL} Case Identifier │ {case_id:<52}{BOX_DOUBLE_VERTICAL}") 

99 lines.append(f"{BOX_DOUBLE_VERTICAL} Classification │ {classification:<52}{BOX_DOUBLE_VERTICAL}") 

100 lines.append(f"{BOX_DOUBLE_VERTICAL} Generation Date │ {generation_date:<52}{BOX_DOUBLE_VERTICAL}") 

101 lines.append(f"{BOX_DOUBLE_VERTICAL}{' ' * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_VERTICAL}") 

102 

103 # Double-line outer box footer 

104 lines.append(f"{BOX_DOUBLE_BOTTOM_LEFT}{BOX_DOUBLE_HORIZONTAL * (STANDARD_WIDTH - 2)}{BOX_DOUBLE_BOTTOM_RIGHT}") 

105 lines.append("") 

106 

107 return "\n".join(lines)
