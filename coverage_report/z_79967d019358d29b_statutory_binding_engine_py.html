301 def bind_violation_to_statutes( 

302 self, 

303 violation_id: str, 

304 violation_type: str, 

305 violation_details: Dict[str, Any] 

306 ) -> StatutoryBinding: 

307 """ 

308 Bind a violation to applicable statutes. 

309  

310 Args: 

311 violation_id: Unique violation identifier 

312 violation_type: Type of violation (must map to statute) 

313 violation_details: Details of the violation for context 

314  

315 Returns: 

316 StatutoryBinding with complete legal framework 

317 """ 

318 # Get applicable statutes 

319 statutes = self.VIOLATION_TO_STATUTE_MAP.get( 

320 violation_type, 

321 self.VIOLATION_TO_STATUTE_MAP["UNKNOWN"] 

322 ) 

323 

324 # Determine enforcement pathway 

325 agencies = list(set([s.enforcement_agency for s in statutes])) 

326 if len(agencies) == 1: 

327 enforcement_pathway = agencies[0].value 

328 else: 

329 enforcement_pathway = "MULTIPLE" 

330 

331 # Generate plain-language explanation 

332 explanation = self._generate_plain_language_explanation( 

333 violation_type, 

334 statutes, 

335 violation_details 

336 ) 

337 

338 # Generate recommended actions 

339 recommended_actions = self._generate_recommended_actions( 

340 violation_type, 

341 statutes 

342 ) 

343 

344 # Generate evidence requirements 

345 evidence_requirements = self._generate_evidence_requirements( 

346 violation_type, 

347 statutes 

348 ) 

349 

350 # Calculate confidence based on evidence strength 

351 confidence = violation_details.get('confidence', 0.85) 

352 

353 binding = StatutoryBinding( 

354 binding_id=f"BIND_{violation_id}", 

355 violation_id=violation_id, 

356 violation_type=violation_type, 

357 statutes=statutes, 

358 confidence=confidence, 

359 enforcement_pathway=enforcement_pathway, 

360 plain_language_explanation=explanation, 

361 recommended_actions=recommended_actions, 

362 evidence_requirements=evidence_requirements 

363 ) 

364 

365 return binding

367 def bind_all_violations( 

368 self, 

369 violations: List[Dict[str, Any]] 

370 ) -> List[StatutoryBinding]: 

371 """ 

372 Bind all violations to statutes. 

373  

374 Args: 

375 violations: List of violation dictionaries 

376  

377 Returns: 

378 List of StatutoryBinding objects 

379 """ 

380 bindings = [] 

381 

382 self.logger.info(f"Binding {len(violations)} violations to statutes...") 

383 

384 for violation in violations: 

385 try: 

386 violation_id = violation.get('violation_id', violation.get('id', 'UNKNOWN')) 

387 violation_type = violation.get('violation_type', violation.get('type', 'UNKNOWN')) 

388 

389 binding = self.bind_violation_to_statutes( 

390 violation_id, 

391 violation_type, 

392 violation 

393 ) 

394 

395 bindings.append(binding) 

396 

397 except Exception as e: 

398 self.logger.warning(f"Failed to bind violation {violation_id}: {e}") 

399 continue 

400 

401 self.logger.info(f"✓ Successfully bound {len(bindings)} violations to statutes") 

402 

403 return bindings

562 def get_enforcement_summary( 

563 self, 

564 bindings: List[StatutoryBinding] 

565 ) -> Dict[str, Any]: 

566 """Generate enforcement pathway summary.""" 

567 

568 summary = { 

569 "total_violations": len(bindings), 

570 "by_agency": { 

571 "SEC": 0, 

572 "DOJ": 0, 

573 "IRS": 0, 

574 "MULTIPLE": 0 

575 }, 

576 "by_case_type": { 

577 "CIVIL": 0, 

578 "CRIMINAL": 0, 

579 "BOTH": 0 

580 }, 

581 "high_confidence_violations": 0, 

582 "criminal_exposure": False, 

583 "statutes_invoked": [] 

584 } 

585 

586 statute_codes = set() 

587 

588 for binding in bindings: 

589 # Count by agency 

590 agency = binding.enforcement_pathway 

591 if agency in summary["by_agency"]: 

592 summary["by_agency"][agency] += 1 

593 

594 # Count by case type and check for criminal 

595 for statute in binding.statutes: 

596 if statute.case_type in [CaseType.CRIMINAL, CaseType.BOTH]: 

597 summary["criminal_exposure"] = True 

598 

599 case_type_str = statute.case_type.value 

600 if case_type_str in summary["by_case_type"]: 

601 summary["by_case_type"][case_type_str] += 1 

602 

603 statute_codes.add(statute.code) 

604 

605 # Count high confidence 

606 if binding.confidence >= 0.85: 

607 summary["high_confidence_violations"] += 1 

608 

609 summary["statutes_invoked"] = sorted(list(statute_codes)) 

610 summary["unique_statutes_count"] = len(statute_codes) 

611 

612 return summary
