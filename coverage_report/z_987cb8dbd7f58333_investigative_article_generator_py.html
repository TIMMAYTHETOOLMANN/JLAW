1""" 

2Investigative Article Generator 

3================================= 

4 

5Synthesises forensic analysis findings into a compelling Wall Street Journal-style 

6investigative narrative — the "story" that turns raw data into a publishable exposé. 

7 

8The engine produces: 

9- A feature-length investigative article (Markdown / HTML) 

10- An executive editor's summary / headline block 

11- Sourced pull-quotes and sidebar call-outs 

12- A "receipts" section linking every claim to its evidence anchor 

13- A recommendations & enforcement action section 

14 

15The tone mirrors investigative financial journalism (WSJ, Bloomberg Businessweek, 

16Financial Times) — declarative, non-hedging, evidence-driven prose that would 

17serve equally well as a deposition exhibit or front-page investigation. 

18"""

49class InvestigativeArticle: 

50 """ 

51 A complete WSJ-style investigative article generated from forensic findings. 

52 

53 Contains a headline block, byline, deck, executive summary, multiple body 

54 sections, a "receipts" evidence section, and an enforcement recommendation. 

55 """ 

56 article_id: str 

57 company_name: str 

58 cik: str 

59 analysis_period: str 

60 generated_at: str 

61 

62 # Headline block 

63 headline: str 

64 sub_headline: str 

65 deck: str # One-sentence "elevator pitch" of the story 

66 

67 # Metadata 

68 word_count: int = 0 

69 severity_label: str = "HIGH" # CRITICAL / HIGH / MEDIUM 

70 

71 # Content 

72 sections: List[ArticleSection] = field(default_factory=list) 

73 receipts: List[Dict[str, Any]] = field(default_factory=list) 

74 enforcement_recommendation: str = "" 

75 key_statistics: Dict[str, Any] = field(default_factory=dict)

78 d = asdict(self)

102 gen.export_markdown(article, output_dir / "investigative_article.md") 

103 gen.export_html(article, output_dir / "investigative_article.html") 

104 """

111 def generate( 

112 self, 

113 company_name: str, 

114 cik: str, 

115 analysis_results: Dict[str, Any], 

116 analysis_period: str = "", 

117 discrepancy_report: Optional[Any] = None, 

118 ) -> InvestigativeArticle: 

119 """ 

120 Generate a full investigative article from forensic findings. 

121 

122 Args: 

123 company_name: Company display name. 

124 cik: SEC CIK. 

125 analysis_results: Full forensic analysis results dict. 

126 analysis_period: Period label (e.g. "FY 2019"). 

127 discrepancy_report: Optional PublicDiscrepancyReport to weave in. 

128 

129 Returns: 

130 InvestigativeArticle dataclass ready for export. 

131 """ 

132 article_id = hashlib.sha256( 

133 f"{cik}{company_name}{datetime.utcnow().isoformat()}".encode() 

134 ).hexdigest()[:16].upper() 

135 

136 violations = analysis_results.get("violations", []) 

137 transactions = analysis_results.get("transactions", []) 

138 beneficiaries = analysis_results.get("beneficiaries", []) 

139 actors = analysis_results.get("actors", []) 

140 total_v = analysis_results.get("total_violations", len(violations)) 

141 critical = analysis_results.get("critical_alerts", 0) 

142 high = analysis_results.get("high_alerts", 0) 

143 

144 # Aggregate penalty 

145 total_penalty = sum(v.get("estimated_penalty", 0) for v in violations) 

146 

147 # Top insider 

148 top_insider = self._top_insider(beneficiaries, violations) 

149 top_profit = self._top_profit(beneficiaries) 

150 

151 # Severity label 

152 if critical > 0: 

153 severity_label = "CRITICAL" 

154 elif high > 5: 

155 severity_label = "HIGH" 

156 else: 

157 severity_label = "MEDIUM" 

158 

159 # Build key statistics 

160 key_stats = { 

161 "total_violations": total_v, 

162 "critical_violations": critical, 

163 "high_violations": high, 

164 "total_penalty_exposure_usd": total_penalty, 

165 "insiders_identified": len( 

166 {v.get("reporting_owner") or v.get("actor") for v in violations} - {""} 

167 ), 

168 "top_insider": top_insider, 

169 "top_insider_profit_usd": top_profit, 

170 "filings_analyzed": len( 

171 {v.get("accession_number") for v in violations} - {""} 

172 ), 

173 } 

174 

175 # Build headline block 

176 headline, sub_headline, deck = self._build_headlines( 

177 company_name, key_stats, severity_label, analysis_period 

178 ) 

179 

180 article = InvestigativeArticle( 

181 article_id=article_id, 

182 company_name=company_name, 

183 cik=cik, 

184 analysis_period=analysis_period or "N/A", 

185 generated_at=datetime.utcnow().isoformat() + "Z", 

186 headline=headline, 

187 sub_headline=sub_headline, 

188 deck=deck, 

189 severity_label=severity_label, 

190 key_statistics=key_stats, 

191 ) 

192 

193 # Build article sections 

194 article.sections = self._build_sections( 

195 company_name, 

196 cik, 

197 analysis_results, 

198 key_stats, 

199 analysis_period, 

200 discrepancy_report, 

201 ) 

202 

203 # Build receipts 

204 article.receipts = self._build_receipts(violations, transactions, beneficiaries) 

205 

206 # Enforcement recommendation 

207 article.enforcement_recommendation = self._build_enforcement_rec( 

208 key_stats, severity_label 

209 ) 

210 

211 # Word count 

212 total_words = sum( 

213 len((s.heading + " " + s.body).split()) 

214 for s in article.sections 

215 ) 

216 article.word_count = total_words 

217 

218 self.logger.info( 

219 "Investigative article generated: %d sections, ~%d words, severity=%s", 

220 len(article.sections), 

221 total_words, 

222 severity_label, 

223 ) 

224 return article

242 def export_json(self, article: InvestigativeArticle, path: Path) -> Path: 

243 """Write the article metadata and structured content as JSON.""" 

244 path.parent.mkdir(parents=True, exist_ok=True) 

245 with open(path, "w", encoding="utf-8") as fh: 

246 json.dump(article.to_dict(), fh, indent=2, default=str) 

247 self.logger.info("Investigative article (JSON) → %s", path) 

248 return path

252 def _top_insider( 

253 self, 

254 beneficiaries: List[Dict[str, Any]], 

255 violations: List[Dict[str, Any]], 

256 ) -> str: 

257 if beneficiaries: 

258 best = max( 

259 beneficiaries, 

260 key=lambda b: b.get("total_profit", b.get("total_shares", 0)), 

261 ) 

262 return best.get("name", "Unknown") 

263 owners = [ 

264 v.get("reporting_owner") or v.get("actor", "") 

265 for v in violations 

266 if v.get("reporting_owner") or v.get("actor") 

267 ] 

268 return owners[0] if owners else "Unknown"

271 if not beneficiaries: 

272 return 0.0 

273 return max( 

274 (b.get("total_profit", 0) for b in beneficiaries), default=0.0

482 top = stats["top_insider"] 

483

505 def _prose_players( 

506 self, 

507 company: str, 

508 beneficiaries: List[Dict[str, Any]], 

509 violations: List[Dict[str, Any]], 

510 stats: Dict[str, Any], 

511 ) -> str: 

512 top = stats["top_insider"] 

513 profit = stats["top_insider_profit_usd"] 

514 

515 # Build insider roll-call 

516 insider_paras: List[str] = [] 

517 

518 if beneficiaries: 

519 for b in beneficiaries[:5]: 

520 name = b.get("name", "Unknown") 

521 role = b.get("role", "Executive") 

522 p = b.get("total_profit", 0) 

523 vcount = b.get("violations", 0) 

524 risk = b.get("risk_score", 0) 

525 insider_paras.append( 

526 f"**{name}** ({role}): ${p:,.0f} in identified equity proceeds " 

527 f"across {vcount} flagged transaction(s). Risk score: {risk}/100." 

528 ) 

529 else: 

530 # Fall back to violation data 

531 by_insider: Dict[str, int] = {} 

532 for v in violations: 

533 name = v.get("reporting_owner") or v.get("actor") or "Unknown" 

534 by_insider[name] = by_insider.get(name, 0) + 1 

535 for name, count in sorted( 

536 by_insider.items(), key=lambda x: -x[1] 

537 )[:5]: 

538 insider_paras.append( 

539 f"**{name}**: {count} flagged transaction(s)." 

540 ) 

541 

542 roster = "\n\n".join(insider_paras) if insider_paras else "See violation log." 

543 

544 return ( 

545 f"Every financial scandal has a cast. Understanding who stood to " 

546 f"gain — and by how much — is the starting point for any enforcement action.\n\n" 

547 f"{roster}\n\n" 

548 f"At the apex of this network stands {top}, whose aggregate equity activity " 

549 f"during the analysis period {"generated approximately ${:,.0f} in identified proceeds. ".format(profit) if profit else "warrants immediate further investigation. "}" 

550 f"The convergence of transaction volume, filing timing, and proximity to " 

551 f"material corporate events places {top.split()[0] if top else 'this individual'} " 

552 f"squarely in the crosshairs of any serious regulatory inquiry.\n\n" 

553 f"These are not passive shareholders. These are insiders with privileged " 

554 f"access to non-public information — and the trading records show they " 

555 f"used it." 

556 )

560 company: str,

566 v for v in violations

568 ]

577 f"{worst.get('days_late', 'N/A')} days — a period during which " 

578 f"{worst.get('reporting_owner', 'the insider')} held information " 

579 f"the public was legally entitled to receive. Under Section 16(a) of " 

580 f"the Securities Exchange Act, Form 4 must be filed within two business " 

581 f"days of a reportable transaction. Filing it late is not a paperwork " 

582 f"error. It is a violation." 

583 ) 

584 

585 return ( 

586 f"The timeline of insider transactions during {company}'s {stats.get('period', 'analysis')} " 

587 f"period is striking not for any single trade, but for the cumulative " 

588 f"pattern it reveals.\n\n" 

589 f"Of the {n_v} violations identified, {n_late} involved late SEC disclosure " 

590 f"of reportable transactions — a statutory obligation that exists specifically " 

591 f"to give the investing public timely notice of insider activity. " 

592 f"Every day of delay is a day the market operates without information " 

593 f"that insiders already possess." 

594 f"{late_para}\n\n" 

595 f"The transactions themselves reveal a consistent directional bias: " 

596 f"disposals concentrated in periods preceding material corporate events, " 

597 f"acquisitions clustered around grant dates that appear optimised for " 

598 f"post-announcement price appreciation. The data does not support the " 

599 f"narrative of passive, long-term holding." 

600 )
