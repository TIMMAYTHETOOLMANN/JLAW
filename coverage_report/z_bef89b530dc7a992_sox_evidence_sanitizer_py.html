71 @classmethod 

72 def sanitize_evidence_text(cls, raw_text: str) -> str: 

73 """Strip HTML/CSS from evidence text fields, extracting clean readable text.""" 

74 if not raw_text or not isinstance(raw_text, str): 

75 return raw_text or '' 

76 

77 clean = raw_text 

78 # Remove HTML tags 

79 clean = re.sub(r'<[^>]+>', ' ', clean) 

80 # Remove CSS properties 

81 clean = re.sub(r'[a-z-]+\s*:\s*[^;]+;', ' ', clean) 

82 # Remove HTML entities 

83 clean = re.sub(r'&[a-z]+;', ' ', clean) 

84 clean = re.sub(r'&#x[0-9a-f]+;', ' ', clean, flags=re.IGNORECASE) 

85 # Remove URLs 

86 clean = re.sub(r'https?://[^\s"]+', '', clean) 

87 # Remove hex color codes 

88 clean = re.sub(r'#[0-9a-fA-F]{3,8}', '', clean) 

89 # Collapse whitespace 

90 clean = re.sub(r'\s+', ' ', clean).strip() 

91 # Remove leading fragments before first uppercase letter 

92 clean = re.sub(r'^[^A-Z]*', '', clean) 

93 

94 return clean or '[Evidence text requires manual extraction from source filing]'

96 @classmethod 

97 def expand_truncated_description(cls, violation: dict) -> str: 

98 """Expand truncated violation descriptions to full canonical text.""" 

99 vtype = violation.get('violation_type', '') 

100 template = cls.CANONICAL_DESCRIPTIONS.get(vtype) 

101 

102 if not template: 

103 return violation.get('description', '') 

104 

105 # Format template with available data 

106 cert_title = '' 

107 affected = violation.get('affected_certifications', []) 

108 if affected: 

109 cert_title = affected[0] 

110 

111 area = 'unspecified' 

112 desc = violation.get('description', '') 

113 if 'revenue_recognition' in desc: 

114 area = 'revenue recognition' 

115 

116 return template.format(cert_title=cert_title, area=area)

118 @classmethod 

119 def fix_certifier_names(cls, sox_result: dict, cik: str) -> dict: 

120 """Fix certifier names that show 'font style' instead of actual names.""" 

121 certifiers = cls.KNOWN_CERTIFIERS.get(str(cik)) 

122 if not certifiers: 

123 return sox_result 

124 

125 # Fix Section 302 certifications 

126 for cert in sox_result.get('section_302_certifications', []): 

127 if cert.get('certifier') == 'font style': 

128 if 'Executive' in cert.get('title', '') or 'CEO' in cert.get('title', ''): 

129 cert['certifier'] = certifiers['ceo']['name'] 

130 cert['title'] = certifiers['ceo']['title'] 

131 cert['_corrected'] = True 

132 cert['_correction_source'] = 'NKE_DEF14A_FY2019' 

133 elif 'Financial' in cert.get('title', '') or 'CFO' in cert.get('title', ''): 

134 cert['certifier'] = certifiers['cfo']['name'] 

135 cert['title'] = certifiers['cfo']['title'] 

136 cert['_corrected'] = True 

137 cert['_correction_source'] = 'NKE_DEF14A_FY2019' 

138 

139 # Sanitize all evidence_text fields in violations 

140 for v in sox_result.get('violations', []): 

141 if v.get('evidence_text'): 

142 v['evidence_text_raw'] = v['evidence_text'] 

143 v['evidence_text'] = cls.sanitize_evidence_text(v['evidence_text']) 

144 # Expand truncated descriptions 

145 v['description'] = cls.expand_truncated_description(v) 

146 

147 # Fix material weakness descriptions 

148 for mw in sox_result.get('material_weaknesses', []): 

149 mw['description_raw'] = mw.get('description', '') 

150 mw['description'] = cls.sanitize_evidence_text(mw.get('description', '')) 

151 if len(mw['description']) < 20: 

152 area = mw.get('control_area', 'unspecified') 

153 mw['description'] = cls.MW_DESCRIPTIONS.get(area, cls.MW_DESCRIPTIONS['unspecified']) 

154 

155 return sox_result
