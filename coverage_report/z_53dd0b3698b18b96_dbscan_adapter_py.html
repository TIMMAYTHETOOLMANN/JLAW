1""" 

2DBSCAN Temporal Clustering Adapter 

3=================================== 

4 

5Density-based spatial clustering adapter for temporal transaction analysis. 

6 

7Implements DBSCAN clustering algorithm (Section 5.3.2) to identify temporally 

8proximate transaction groups indicative of coordinated structuring. 

9 

10Regulatory Basis: 

11 - Same-day clustering suggests deliberate fragmentation per SEC Release 34-46421 

12 - Multi-day clustering within eps_days suggests coordinated execution 

13 

14Reference: 

15 - Section 5.3.2: DBSCAN Clustering Algorithm 

16"""

28def detect_temporal_clusters( 

29 distance_matrix: np.ndarray, 

30 transactions: List[Transaction], 

31 eps_days: int = 1, 

32 min_samples: int = 2 

33) -> List[TransactionCluster]: 

34 """ 

35 Apply density-based clustering to identify temporally proximate transaction groups. 

36  

37 Uses DBSCAN (Density-Based Spatial Clustering of Applications with Noise) 

38 algorithm to detect clusters of transactions occurring within eps_days of 

39 each other, requiring at least min_samples transactions per cluster. 

40  

41 Regulatory Basis: 

42 - Same-day clustering suggests deliberate fragmentation per SEC Release 34-46421 

43 - Multi-day clustering within eps_days suggests coordinated execution 

44  

45 Args: 

46 distance_matrix: Symmetric pairwise temporal distance matrix (n x n) 

47 where M[i][j] = days between transaction i and j 

48 transactions: List of Transaction objects corresponding to matrix indices 

49 eps_days: Maximum temporal distance (in days) between transactions in cluster. 

50 Default 1 (same-day clustering) 

51 min_samples: Minimum number of transactions required to form a cluster. 

52 Default 2 (pairs of transactions) 

53  

54 Returns: 

55 List of TransactionCluster objects, one per detected cluster. 

56 Transactions labeled as noise (-1) are excluded from output. 

57  

58 Example: 

59 >>> distance_matrix = np.array([[0, 0, 5], [0, 0, 5], [5, 5, 0]]) 

60 >>> clusters = detect_temporal_clusters(distance_matrix, transactions, eps_days=1) 

61 >>> len(clusters) # Two transactions within 1 day 

62 1 

63 """ 

64 # Apply DBSCAN clustering with precomputed distance matrix 

65 clustering = DBSCAN(eps=eps_days, min_samples=min_samples, metric='precomputed') 

66 labels = clustering.fit_predict(distance_matrix) 

67 

68 # Group transactions by cluster label, excluding noise (-1) 

69 cluster_dict: Dict[int, List[Transaction]] = {} 

70 for idx, label in enumerate(labels): 

71 if label != -1: # Exclude noise points 

72 if label not in cluster_dict: 

73 cluster_dict[label] = [] 

74 cluster_dict[label].append(transactions[idx]) 

75 

76 # Create TransactionCluster objects 

77 clusters = [] 

78 for cluster_id, cluster_transactions in cluster_dict.items(): 

79 if len(cluster_transactions) < min_samples: 

80 continue # Skip clusters below minimum size 

81 

82 # Extract cluster metadata 

83 dates = [t.transaction_date for t in cluster_transactions] 

84 start_date = min(dates) 

85 end_date = max(dates) 

86 

87 # Calculate aggregate metrics 

88 total_shares = sum(abs(t.shares) for t in cluster_transactions) 

89 zero_dollar_count = sum(1 for t in cluster_transactions if t.is_zero_dollar) 

90 

91 # Get reporting person info (should be same across cluster) 

92 reporting_person_cik = cluster_transactions[0].reporting_person_cik 

93 reporting_person_name = cluster_transactions[0].reporting_person_name 

94 

95 # Generate cluster identifier 

96 cluster_hash = _generate_cluster_id( 

97 reporting_person_cik, 

98 start_date, 

99 end_date, 

100 len(cluster_transactions) 

101 ) 

102 

103 # Initial cluster score (will be refined by scoring module) 

104 cluster_score = 0.0 

105 

106 cluster = TransactionCluster( 

107 cluster_id=cluster_hash, 

108 reporting_person_cik=reporting_person_cik, 

109 reporting_person_name=reporting_person_name, 

110 transactions=cluster_transactions, 

111 start_date=start_date, 

112 end_date=end_date, 

113 total_shares=total_shares, 

114 zero_dollar_count=zero_dollar_count, 

115 cluster_score=cluster_score, 

116 detection_timestamp=datetime.utcnow() 

117 ) 

118 

119 clusters.append(cluster) 

120 

121 return clusters

124def _generate_cluster_id( 

125 reporting_person_cik: str, 

126 start_date, 

127 end_date, 

128 transaction_count: int 

129) -> str: 

130 """ 

131 Generate unique cluster identifier using SHA-256 hash. 

132  

133 Creates deterministic cluster ID from key attributes for evidence tracking. 

134  

135 Args: 

136 reporting_person_cik: CIK of reporting person 

137 start_date: Earliest transaction date in cluster 

138 end_date: Latest transaction date in cluster 

139 transaction_count: Number of transactions in cluster 

140  

141 Returns: 

142 Hexadecimal SHA-256 hash (first 16 characters) 

143 """ 

144 content = f"{reporting_person_cik}|{start_date.isoformat()}|{end_date.isoformat()}|{transaction_count}" 

145 hash_obj = hashlib.sha256(content.encode('utf-8')) 

146 return hash_obj.hexdigest()[:16]
