1""" 

2MNPI Scoring Algorithm 

3====================== 

4 

5Calculate Material Nonpublic Information (MNPI) inference probability scores 

6for zero-dollar transactions in proximity to material corporate events. 

7 

8Per Section 6.3 of JLAW Zero-Dollar Transaction Forensic Specification v1.0. 

9 

10Reference: 

11 - Section 6.3: Proximity Detection Algorithm 

12 - Section 6.3.1: MNPI Score Calculation Formula 

13 - 15 U.S.C. § 78j(b) - Securities Exchange Act Section 10(b) 

14 - 17 CFR § 240.10b-5 - Rule 10b-5 

15"""

24def calculate_mnpi_score( 

25 event: MaterialEvent, 

26 days_delta: int, 

27 proximity_type: str, 

28 sensitivity: str = None 

29) -> Decimal: 

30 """ 

31 Calculate MNPI inference probability score. 

32  

33 Computes the likelihood that a zero-dollar transaction was executed 

34 based on Material Nonpublic Information (MNPI), considering: 

35 1. Event Sensitivity Level (CRITICAL=1.0, HIGH=0.75, MODERATE=0.5, LOW=0.25) 

36 2. Temporal Proximity (exponential decay with distance from event) 

37 3. Direction Alignment (PRE_EVENT dispositions before negative news score higher) 

38  

39 Formula: 

40 MNPI_SCORE = SENSITIVITY * e^(-λ * days_delta) * DIRECTION_FACTOR 

41  

42 Where: 

43 λ = 0.1 (decay constant, calibrated to half-life of ~7 days) 

44 DIRECTION_FACTOR = 1.0 for dispositions before negative events 

45 = 0.5 for dispositions before positive events 

46 = 0.25 for post-event transactions 

47  

48 Args: 

49 event: MaterialEvent object with sensitivity information 

50 days_delta: Absolute days between transaction and event (always positive) 

51 proximity_type: 'PRE' (before event) or 'POST' (after event) 

52 sensitivity: Optional override for event sensitivity 

53 (CRITICAL, HIGH, MODERATE, LOW, VARIABLE) 

54  

55 Returns: 

56 Decimal: MNPI exploitation probability score [0.0 - 1.0] 

57  

58 Example: 

59 >>> event = MaterialEvent(event_type="2.02", ...) 

60 >>> score = calculate_mnpi_score(event, days_delta=3, proximity_type='PRE', sensitivity='CRITICAL') 

61 >>> print(score) # 0.741 (high probability) 

62 """ 

63 # Sensitivity mapping 

64 sensitivity_map = { 

65 'CRITICAL': Decimal('1.0'), 

66 'HIGH': Decimal('0.75'), 

67 'MODERATE': Decimal('0.5'), 

68 'LOW': Decimal('0.25'), 

69 'VARIABLE': Decimal('0.5'), # Default for variable sensitivity events 

70 } 

71 

72 # Use provided sensitivity or extract from event_type 

73 # Note: event object stores event_type as string, sensitivity must be provided 

74 # or determined from taxonomy 

75 if sensitivity is None: 

76 sensitivity = 'MODERATE' # Default fallback 

77 

78 sensitivity_score = sensitivity_map.get(sensitivity, Decimal('0.25')) 

79 

80 # Exponential decay based on temporal distance 

81 # λ = 0.1 gives half-life of ~7 days 

82 lambda_decay = 0.1 

83 temporal_factor = Decimal(str(math.exp(-lambda_decay * abs(days_delta)))) 

84 

85 # Direction factor - pre-event more suspicious than post-event 

86 if proximity_type == 'PRE': 

87 direction_factor = Decimal('1.0') # Pre-event dispositions are highly suspicious 

88 else: # POST 

89 direction_factor = Decimal('0.25') # Post-event less suspicious 

90 

91 # Calculate final MNPI score 

92 mnpi_score = sensitivity_score * temporal_factor * direction_factor 

93 

94 # Quantize to 3 decimal places 

95 return mnpi_score.quantize(Decimal('0.001'))

98def get_event_citations(event: MaterialEvent, sensitivity: str = None) -> List[str]: 

99 """ 

100 Return applicable regulatory citations for event type. 

101  

102 Provides statutory and regulatory references for enforcement actions 

103 based on event sensitivity and type. 

104  

105 Args: 

106 event: MaterialEvent object 

107 sensitivity: Optional sensitivity level (CRITICAL, HIGH, MODERATE, LOW) 

108  

109 Returns: 

110 List of regulatory citations (CFR sections, USC sections) 

111  

112 Example: 

113 >>> event = MaterialEvent(event_type="2.02", ...) 

114 >>> citations = get_event_citations(event, sensitivity='CRITICAL') 

115 >>> print(citations) 

116 ['15 U.S.C. § 78j(b)', '17 CFR § 240.10b-5', '17 CFR § 240.10b5-1',  

117 '18 U.S.C. § 1348', 'Regulation FD (17 CFR § 243)'] 

118 """ 

119 # Base citations applicable to all insider trading violations 

120 base_citations = [ 

121 '15 U.S.C. § 78j(b)', # Securities Exchange Act Section 10(b) 

122 '17 CFR § 240.10b-5', # Rule 10b-5 (antifraud) 

123 ] 

124 

125 # Add critical-level citations for high-sensitivity events 

126 if sensitivity == 'CRITICAL': 

127 base_citations.extend([ 

128 '17 CFR § 240.10b5-1', # Rule 10b5-1 (trading plans) 

129 '18 U.S.C. § 1348', # Securities fraud (criminal statute) 

130 ]) 

131 

132 # Add Regulation FD for earnings-related events 

133 event_type = event.event_type.lower() if hasattr(event, 'event_type') else '' 

134 if 'earnings' in event_type or 'quarterly' in event_type or 'annual' in event_type: 

135 base_citations.append('Regulation FD (17 CFR § 243)') 

136 

137 # Add Form 8-K citations for specific event types 

138 if event_type in ['2.02', '2.06', '4.02', '1.03', '3.01', '5.01']: 

139 # Major disclosure events 

140 base_citations.append('17 CFR § 229.308 (Item 308 - Internal Controls)') 

141 

142 return base_citations

145def determine_mnpi_severity(mnpi_score: Decimal) -> str: 

146 """ 

147 Classify MNPI score into severity level. 

148  

149 Maps MNPI probability score to investigation priority level. 

150  

151 Args: 

152 mnpi_score: MNPI inference probability [0.0 - 1.0] 

153  

154 Returns: 

155 Severity level: CRITICAL, HIGH, MODERATE, or LOW 

156  

157 Classification: 

158 - CRITICAL: score >= 0.70 (immediate investigation) 

159 - HIGH: score >= 0.50 (priority investigation) 

160 - MODERATE: score >= 0.30 (enhanced monitoring) 

161 - LOW: score < 0.30 (routine monitoring) 

162 """ 

163 if mnpi_score >= Decimal('0.70'): 

164 return 'CRITICAL' 

165 elif mnpi_score >= Decimal('0.50'): 

166 return 'HIGH' 

167 elif mnpi_score >= Decimal('0.30'): 

168 return 'MODERATE' 

169 else: 

170 return 'LOW'
