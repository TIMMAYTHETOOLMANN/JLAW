19@dataclass 

20class Transaction: 

21 """ 

22 SEC Form 4 transaction with forensic metadata. 

23  

24 Represents a single transaction from Form 4 with all required fields 

25 for forensic analysis including zero-dollar detection, late filing 

26 tracking, and notional value calculation. 

27  

28 Attributes: 

29 accession_number: SEC EDGAR accession number (NNNNNNNNNN-NN-NNNNNN) 

30 issuer_cik: Central Index Key of the issuing company 

31 issuer_name: Official company name 

32 reporting_person_cik: CIK of the reporting insider 

33 reporting_person_name: Full name of reporting person 

34 transaction_date: Date transaction was executed 

35 filing_date: Date Form 4 was filed with SEC 

36 transaction_code: SEC Form 4 transaction code (P, S, M, A, etc.) 

37 shares: Number of shares involved in transaction 

38 price_per_share: Transaction price per share (None for zero-dollar) 

39 transaction_acquired_disposed: 'A' (acquired) or 'D' (disposed) 

40 shares_owned_following: Total shares owned after transaction 

41 direct_indirect: 'D' (direct) or 'I' (indirect ownership) 

42 nature_of_ownership: Description if indirect (e.g., "By Trust") 

43 footnotes: List of footnote references 

44 security_title: Type of security (e.g., "Common Stock") 

45 form_type: Filing form type (typically "4") 

46 document_url: URL to filed Form 4 document 

47 derivative_security: Whether this is a derivative security 

48 conversion_exercise_price: Price for derivative conversion/exercise 

49 exercise_date: Date derivative can be exercised 

50 expiration_date: Date derivative expires 

51 """

89 @property 

90 def is_zero_dollar(self) -> bool: 

91 """ 

92 Determine if this is a zero-dollar transaction. 

93  

94 Per Section 2.1: A transaction where price_per_share is None, 

95 zero, or explicitly marked as $0.00 in Form 4 filings. 

96  

97 Returns: 

98 True if transaction has zero or null price 

99 """ 

100 return ( 

101 self.price_per_share is None or 

102 self.price_per_share == Decimal('0') or 

103 self.price_per_share == Decimal('0.00') 

104 ) 

105 

106 @property 

107 def notional_value(self) -> Decimal: 

108 """ 

109 Calculate notional value of the transaction. 

110  

111 For zero-dollar transactions, this requires market price lookup. 

112 For priced transactions, multiply shares by price_per_share. 

113  

114 Returns: 

115 Notional value (shares * price), or 0 for zero-dollar 

116 """ 

117 if self.is_zero_dollar: 

118 return Decimal('0') 

119 return abs(self.shares) * (self.price_per_share or Decimal('0')) 

120 

121 @property 

122 def days_to_filing(self) -> int: 

123 """ 

124 Calculate days between transaction and filing. 

125  

126 Section 16(a) requires filing within 2 business days. 

127 This calculates calendar days for anomaly detection. 

128  

129 Returns: 

130 Number of calendar days from transaction to filing 

131 """ 

132 delta = self.filing_date - self.transaction_date 

133 return delta.days 

134 

135 @property 

136 def is_late_filing(self) -> bool: 

137 """ 

138 Determine if filing is late per Section 16(a). 

139  

140 SEC requires Form 4 within 2 business days. We use 3 calendar 

141 days as conservative threshold for late filing detection. 

142  

143 Returns: 

144 True if filed more than 3 calendar days after transaction 

145 """ 

146 return self.days_to_filing > 3

177@dataclass 

178class TransactionCluster: 

179 """ 

180 Temporal cluster of related transactions. 

181  

182 Groups transactions by reporting person and temporal proximity 

183 for pattern detection. Implements temporal clustering algorithm 

184 per Section 4.2 of the specification. 

185  

186 Attributes: 

187 cluster_id: Unique cluster identifier 

188 reporting_person_cik: CIK of reporting person 

189 reporting_person_name: Name of reporting person 

190 transactions: List of transactions in cluster 

191 start_date: Earliest transaction date in cluster 

192 end_date: Latest transaction date in cluster 

193 total_shares: Sum of all shares in cluster 

194 zero_dollar_count: Count of zero-dollar transactions 

195 cluster_score: Temporal clustering risk score (0-100) 

196 detection_timestamp: When cluster was identified 

197 """ 

198 cluster_id: str 

199 reporting_person_cik: str 

200 reporting_person_name: str 

201 transactions: List[Transaction] 

202 start_date: date 

203 end_date: date 

204 total_shares: Decimal 

205 zero_dollar_count: int 

206 cluster_score: float 

207 detection_timestamp: datetime = field(default_factory=datetime.utcnow) 

208 

209 @property 

210 def cluster_span_days(self) -> int: 

211 """Calculate temporal span of cluster in days.""" 

212 delta = self.end_date - self.start_date 

213 return delta.days 

214 

215 @property 

216 def transaction_count(self) -> int: 

217 """Total number of transactions in cluster.""" 

218 return len(self.transactions) 

219 

220 @property 

221 def zero_dollar_ratio(self) -> float: 

222 """Ratio of zero-dollar to total transactions.""" 

223 if self.transaction_count == 0: 

224 return 0.0 

225 return self.zero_dollar_count / self.transaction_count 

226 

227 @property 

228 def average_notional_value(self) -> Decimal: 

229 """Average notional value across non-zero transactions.""" 

230 non_zero = [t for t in self.transactions if not t.is_zero_dollar] 

231 if not non_zero: 

232 return Decimal('0') 

233 total = sum(t.notional_value for t in non_zero) 

234 return total / len(non_zero) 

235 

236 def to_dict(self) -> Dict[str, Any]: 

237 """Convert to dictionary for serialization.""" 

238 return { 

239 'cluster_id': self.cluster_id, 

240 'reporting_person_cik': self.reporting_person_cik, 

241 'reporting_person_name': self.reporting_person_name, 

242 'transaction_count': self.transaction_count, 

243 'start_date': self.start_date.isoformat(), 

244 'end_date': self.end_date.isoformat(), 

245 'cluster_span_days': self.cluster_span_days, 

246 'total_shares': str(self.total_shares), 

247 'zero_dollar_count': self.zero_dollar_count, 

248 'zero_dollar_ratio': self.zero_dollar_ratio, 

249 'cluster_score': self.cluster_score, 

250 'average_notional_value': str(self.average_notional_value), 

251 'detection_timestamp': self.detection_timestamp.isoformat(),
