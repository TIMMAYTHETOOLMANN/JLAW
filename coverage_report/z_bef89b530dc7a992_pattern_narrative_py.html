1""" 

2MODULE 6: PATTERN NARRATIVE SYNTHESIS ENGINE (DEF-010, DEF-020)

10class PatternNarrativeSynthesizer: 

11 """Generate prosecutorial narrative from enriched analysis results.""" 

12 

13 @classmethod 

14 def extract_narrative_elements(cls, enriched_results: dict) -> dict: 

15 transactions = enriched_results.get('transactions', []) 

16 total_benefit = sum(t.get('economic_benefit', 0) or 0 for t in transactions) 

17 beneficiaries = enriched_results.get('beneficiary_rollup', []) 

18 

19 knight_txns = [ 

20 t for t in transactions 

21 if t.get('transaction_code') == 'J' 

22 and any( 

23 kw in (t.get('reporting_owner') or t.get('insider_name') or '').upper() 

24 for kw in ['KNIGHT', 'SWOOSH'] 

25 ) 

26 ] 

27 knight_transfers = { 

28 'total_shares': sum(abs(t.get('shares', 0) or 0) for t in knight_txns), 

29 'total_value': sum(t.get('economic_benefit', 0) or 0 for t in knight_txns), 

30 'transactions': knight_txns, 

31 } 

32 

33 late_filers = [t for t in transactions if t.get('is_late_filed') or t.get('is_late')] 

34 sox_violations = enriched_results.get('sox_violations', []) 

35 

36 return { 

37 'total_benefit': total_benefit, 

38 'beneficiaries': beneficiaries, 

39 'knight_transfers': knight_transfers, 

40 'late_filers': late_filers, 

41 'sox_violations': sox_violations, 

42 }
