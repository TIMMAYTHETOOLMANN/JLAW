34 def format(exec_summary: Dict[str, Any]) -> str: 

35 """ 

36 Format executive intelligence briefing per specification. 

37  

38 Args: 

39 exec_summary: Executive summary dictionary with: 

40 - threat_level: CRITICAL/HIGH/MEDIUM/LOW 

41 - threat_statement: Primary threat assessment 

42 - total_violations: Total violation count 

43 - critical_violations: Critical violation count 

44 - high_violations: High violation count 

45 - total_actors: Number of actors involved 

46 - key_findings: List of key findings with relevance 

47 - enforcement_recommendation: Recommendation text 

48 - primary_enforcement_agencies: List of agencies 

49 - priority_matrix: Dict of priority tiers 

50  

51 Returns: 

52 Formatted intelligence briefing string 

53 """ 

54 lines = [] 

55 

56 # Section header 

57 lines.append(SUBSECTION_DIVIDER) 

58 lines.append(SECTION_1_TITLE) 

59 lines.append(SUBSECTION_DIVIDER) 

60 lines.append("") 

61 

62 # Threat Assessment Box 

63 lines.extend(ExecutiveBriefingFormatter._format_threat_assessment(exec_summary)) 

64 lines.append("") 

65 

66 # Key Findings at a Glance 

67 lines.extend(ExecutiveBriefingFormatter._format_key_findings(exec_summary)) 

68 lines.append("") 

69 

70 # Investigation Priority Matrix 

71 lines.extend(ExecutiveBriefingFormatter._format_priority_matrix(exec_summary)) 

72 lines.append("") 

73 

74 # Enforcement Recommendation 

75 lines.append(f"{STATUS_ACTION_REQUIRED} ENFORCEMENT RECOMMENDATION:") 

76 lines.append("") 

77 lines.append(exec_summary.get('enforcement_recommendation', 'No recommendation available.')) 

78 lines.append("") 

79 

80 # Primary enforcement agencies 

81 agencies = exec_summary.get('primary_enforcement_agencies', []) 

82 if agencies: 

83 lines.append(f"{STATUS_ACTION_REQUIRED} PRIMARY ENFORCEMENT AGENCIES:") 

84 lines.append("") 

85 for agency in agencies: 

86 lines.append(f" • {agency}") 

87 lines.append("") 

88 

89 return "\n".join(lines)

94 lines = [] 

95 

96 threat_level = exec_summary.get('threat_level', 'UNKNOWN')

213 @staticmethod 

214 def _wrap_text(text: str, width: int) -> List[str]: 

215 """Wrap text to specified width.""" 

216 words = text.split() 

217 lines = [] 

218 current_line = [] 

219 current_length = 0 

220 

221 for word in words: 

222 word_length = len(word) + (1 if current_line else 0) 

223 if current_length + word_length <= width: 

224 current_line.append(word) 

225 current_length += word_length 

226 else: 

227 if current_line: 

228 lines.append(' '.join(current_line)) 

229 current_line = [word] 

230 current_length = len(word) 

231 

232 if current_line: 

233 lines.append(' '.join(current_line)) 

234 

235 return lines
